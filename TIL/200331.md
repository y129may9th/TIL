# PHP
## explode(string $delimiter , string $string, [int limit]) : array
```php
>>> $hoge = "yui, hoge. piyo, fuga"
=> "yui, hoge. piyo, fuga"
>>> explode(',', $hoge )
=> [
     "yui",
     " hoge. piyo",
     " fuga",
   ]

//  limit
>>> explode(',', $hoge, 1)
=> [
     "yui, hoge. piyo, fuga",
   ]
>>>
>>> explode(',', $hoge, 2 )
=> [
     "yui",
     " hoge. piyo, fuga",
   ]
>>> explode(',', $hoge, -1 )
=> [
     "yui",
     " hoge. piyo",
   ]
>>> explode(',', $hoge, -3 )
=> []
>>> explode(',', $hoge, 3 )
=> [
     "yui",
     " hoge. piyo",
     " fuga",
   ]
```
